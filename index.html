// 메뉴 버튼 토글 const menuBtn = document.getElementById('menuBtn'); const submenu = document.getElementById('submenu'); menuBtn.addEventListener('click', () => { submenu.style.display = submenu.style.display === 'flex' ? 'none' : 'flex'; }); // 사이드 패널 토글 const attractionBtn = document.querySelector('.attractions'); const sidePanel = document.getElementById('sidePanel'); attractionBtn.addEventListener('click', () => { sidePanel.classList.toggle('show'); }); // 탭 전환 const tabButtons = document.querySelectorAll('.tabs button'); const tabContents = document.querySelectorAll('.tab-content'); tabButtons.forEach(btn => { btn.addEventListener('click', () => { const tab = btn.getAttribute('data-tab'); tabButtons.forEach(b => b.classList.remove('active')); btn.classList.add('active'); tabContents.forEach(c => { c.classList.remove('active'); if (c.id === tab) c.classList.add('active'); }); }); }); // 명소 클릭 이벤트 document.querySelectorAll('.attraction-item').forEach(item => { item.addEventListener('click', () => { const lat = item.getAttribute('data-lat'); const lng = item.getAttribute('data-lng'); map.setView([lat, lng], 16); L.popup() .setLatLng([lat, lng]) .setContent(`${item.textContent}`) .openOn(map); }); }); // 현재 위치 버튼 이벤트 const locationBtn = document.querySelector('.location'); locationBtn.addEventListener('click', () => { if (navigator.geolocation) { navigator.geolocation.getCurrentPosition(pos => { const lat = pos.coords.latitude; const lng = pos.coords.longitude; map.setView([lat, lng], 16); L.marker([lat, lng]).addTo(map) .bindPopup("현재 위치") .openPopup(); }, err => { alert("현재 위치를 가져올 수 없습니다."); }); } else { alert("이 브라우저에서는 위치 정보를 지원하지 않습니다."); } }); // 아코디언 동작 const accordionHeaders = document.querySelectorAll('.accordion-header'); accordionHeaders.forEach(header => { header.addEventListener('click', () => { const body = header.nextElementSibling; // 다른 아코디언 닫기 document.querySelectorAll('.accordion-body').forEach(b => { if (b !== body) b.classList.remove('show'); }); // 현재 아코디언 토글 body.classList.toggle('show'); }); });
